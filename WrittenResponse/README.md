```csharp
public int CountLetter(char guess)
{
    if (!char.IsLetter(guess))
    {
        throw new ArgumentException($"Invalid character: {guess}.");
    }

    guess = char.ToUpper(guess);
    int count;
    count = 0;
    foreach (char c in this.fullWord)
    {
        if (c == guess)
        {
            count++;
        }
    }

    return count;
}
```
